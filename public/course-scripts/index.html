Terrain Meshgenerierung #1

 1@tool # ermöglicht es Skripte im Editor (ohne dass das Spiel gestartet werden muss) auszuführen
 2extends MeshInstance3D
 3
 4@export var width = 1.0
 5@export var height = 1.0
 6@export var noise_frequency = 0.5
 7@export var noise_amplitude = 0.1
 8@export var subdivisions = 10
 9
10@export var regenerate_mesh: bool:
11 set(value):
12  create_mesh()
13
14var noise = FastNoiseLite.new()
15
16func create_mesh():
17 noise.seed = randi()
18 var arrmesh = ArrayMesh.new()
19 var arrays = []
20 
21 arrays.resize(Mesh.ARRAY_MAX)
22 
23 var verts = PackedVector3Array()
24 
25 var cellsize_x = width / subdivisions
26 var cellsize_z = height / subdivisions
27 
28 for x in subdivisions:
29  var p_row = x * cellsize_x
30  for z in subdivisions:
31   var p_column = z * cellsize_z
32
33   # Wir generieren die vier Eckpunkte einer Zelle...
34   var p1 = Vector3(p_row, height_map(p_row, p_column), p_column)
35   var p2 = Vector3(p_row + cellsize_x,  height_map(p_row + cellsize_x, p_column), p_column)
36   var p3 = Vector3(p_row,  height_map(p_row, p_column + cellsize_z), p_column + cellsize_z)
37   var p4 = Vector3(p_row + cellsize_x,  height_map(p_row + cellsize_x, p_column + cellsize_z), p_column + cellsize_z)
38
39   #... und übergeben diese dem verts - array um zwei dreiecke zu bilden.
40   # zwei der Eckpunkte werden hier daher doppelt verwendet 
41   verts.append_array([p1, p2, p3, p2, p4, p3])
42 
43 arrays[Mesh.ARRAY_VERTEX] = verts
44 
45 arrmesh.add_surface_from_arrays(Mesh.PRIMITIVE_TRIANGLES, arrays)
46 mesh = arrmesh
47
48
49func height_map(x: float, z: float) -> float:
50 noise.fractal_octaves = 5
51 noise.frequency = noise_frequency
52 return noise.get_noise_2d(x, z) * noise_amplitude
